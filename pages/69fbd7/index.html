# ф║Лф╗╢х╛кчОпя╝Мш┤Яш┤гхдДчРЖцЙАцЬЙф║Лф╗╢
function eventLoop():
    while server is running:
        processTimers()                  # хдДчРЖхоЪцЧ╢ф╗╗хКбя╝ИхжВш┐ЗцЬЯ keyуАБAOF цМБф╣ЕхМЦчнЙя╝Й
        acceptClientConnections()        # цОехПЧцЦ░чЪДховцИ╖члпш┐ЮцОе
        processClientRequests()          # хдДчРЖховцИ╖члпхС╜ф╗дшп╖ц▒В
        handleReplication()              # хдДчРЖф╕╗ф╗ОхдНхИ╢цХ░цНохРМцне
        handleSentinel()                 # хдДчРЖхУихЕ╡чЫСцОзхТМцХЕщЪЬхИЗцНв
        handleCluster()                  # щЫЖч╛дцибх╝Пф╕Ля╝МхдДчРЖхИЖчЙЗш╖пчФ▒хТМцХ░цНош┐Бчз╗

# хдДчРЖхоЪцЧ╢ф╗╗хКбя╝МхжВш┐ЗцЬЯ keyуАБцМБф╣ЕхМЦ
function processTimers():
    expireKeysIfNeeded()                 # цгАцЯех╣╢хИащЩдш┐ЗцЬЯчЪДщФо
    handleAOFAndRDBPersistence()         # ца╣цНочнЦчХешзжхПС AOF хТМ RDB цМБф╣ЕхМЦ
    evictKeysIfNeeded()                  # хжВцЮЬхЖЕхнШш╢ЕхЗ║ф╕КщЩРя╝МшзжхПСц╖Шц▒░чнЦчХе

# хдДчРЖцЦ░ховцИ╖члпш┐ЮцОе
function acceptClientConnections():
    when new client connects:
        client = createClient()          # хИЫх╗║ховцИ╖члпхп╣ш▒б
        registerReadEvent(client)        # ц│ихЖМшп╗ф║Лф╗╢я╝МчЫСхРмховцИ╖члпшп╖ц▒В

# хдДчРЖховцИ╖члпхС╜ф╗дшп╖ц▒В
function processClientRequests():
    for each client in server.clients:
        if client has data to read:
            readDataFromClient(client)   # ф╗О socket ф╕ншп╗хПЦшп╖ц▒ВцХ░цНо
            if data is valid:
                command = parseCommand(client)  # шзгцЮР Redis хС╜ф╗д
                executeCommand(client, command) # цЙзшбМхС╜ф╗д

# цЙзшбМ Redis хС╜ф╗д
function executeCommand(client, command):
    if command is valid:
        cmdFunction = lookupCommand(command)  # цЯецЙ╛хС╜ф╗дхЗ╜цХ░
        result = cmdFunction(client)          # цЙзшбМхС╜ф╗д
        addToClientOutputBuffer(client, result) # х░Жч╗УцЮЬхЖЩхЕеш╛УхЗ║ч╝УхЖ▓хМ║
        replicateCommandToSlaves(command)     # хРМцнехС╜ф╗дхИ░ф╗ОцЬНхКбхЩия╝Иф╕╗ф╗ОхдНхИ╢я╝Й

        # хЬихС╜ф╗дцЙзшбМш┐ЗчиЛф╕ня╝МхПпшГ╜шзжхПСцФпч║┐цУНф╜Ь
        checkKeyExpiration(client, command)  # цгАцЯе key цШпхРжш┐ЗцЬЯя╝Мш┐ЫшбМцГ░цАзхИащЩд
        handlePersistence(command)           # ца╣цНощЕНч╜оя╝МшзжхПС AOF цИЦ RDB цМБф╣ЕхМЦ
        evictKeysIfNeeded()                  # х╜УхЖЕхнШф╕Нш╢│цЧ╢я╝МшзжхПСц╖Шц▒░чнЦчХе

    else:
        sendError(client, "Invalid command") # ш┐ФхЫЮщФЩшппц╢ИцБп

# ш┐ФхЫЮхУНх║Фч╗ЩховцИ╖члп
function addToClientOutputBuffer(client, result):
    client.outputBuffer.append(result)       # х░ЖхС╜ф╗дцЙзшбМч╗УцЮЬц╖╗хКахИ░ш╛УхЗ║ч╝УхЖ▓хМ║
    registerWriteEvent(client)               # ц│ихЖМхЖЩф║Лф╗╢я╝МхЗЖхдЗхПСщАБч╗ЩховцИ╖члп

# хПСщАБхУНх║Ф
function sendResponseToClient(client):
    writeDataToSocket(client, client.outputBuffer)  # х░ЖцХ░цНохПСщАБч╗ЩховцИ╖члп
    if write success:
        clearClientOutputBuffer(client)      # ц╕Ечй║ч╝УхЖ▓хМ║
    else:
        handleSocketError(client)            # хдДчРЖ socket щФЩшпп

# цФпч║┐я╝ЪцгАцЯе key ш┐ЗцЬЯя╝Мш┐ЗцЬЯчнЦчХе
function checkKeyExpiration(client, command):
    key = getKeyFromCommand(command)
    if isKeyExpired(key):
        deleteKey(client.db, key)            # цГ░цАзхИащЩдш┐ЗцЬЯчЪД key

function expireKeysIfNeeded():
    for each db in server.databases:
        for each key in db.keys:
            if isKeyExpired(key):
                deleteKey(db, key)           # ф╕╗хКихИащЩдш┐ЗцЬЯчЪД key

# цФпч║┐я╝ЪцМБф╣ЕхМЦя╝МхдДчРЖ RDB хТМ AOF щА╗ш╛С
function handlePersistence(command):
    if AOF is enabled:
        writeCommandToAOF(command)           # х░ЖхС╜ф╗дхЖЩхЕе AOF цЦЗф╗╢
    if RDB snapshot is needed:
        saveRDBSnapshot()                    # шзжхПС RDB х┐лчЕзчФЯцИР

function handleAOFAndRDBPersistence():
    if timeToSaveRDB():
        saveRDB()                            # цМБф╣ЕхМЦ RDB х┐лчЕз
    if timeToRewriteAOF():
        rewriteAOF()                         # шзжхПС AOF щЗНхЖЩф╗ехОЛч╝йцЧех┐Ч

# цФпч║┐я╝Ъц╖Шц▒░чнЦчХея╝МLRUуАБLFU
function evictKeysIfNeeded():
    while memoryUsageExceedsLimit():
        candidates = sampleKeys()            # щЪПцЬ║щЗЗца╖ key
        keyToEvict = findLRUOrLFUKey(candidates) # ца╣цНо LRU/LFU чнЦчХещАЙцЛйц╖Шц▒░ key
        deleteKey(db, keyToEvict)            # хИащЩд key ф╗ещЗКцФ╛хЖЕхнШ

# цФпч║┐я╝Ъф╕╗ф╗ОхдНхИ╢я╝Мф╕╗цЬНхКбхЩих░ЖхС╜ф╗дхРМцнеч╗Щф╗ОцЬНхКбхЩи
function replicateCommandToSlaves(command):
    if server.isMaster():
        for each slave in server.slaves:
            sendCommandToSlave(slave, command)  # хдНхИ╢хС╜ф╗дхИ░ф╗ОцЬНхКбхЩи

function handleReplication():
    if server.isMaster():
        sendReplicationDataToSlaves()          # ф╕╗цЬНхКбхЩихРСф╗ОцЬНхКбхЩихРМцнецХ░цНо
    if server.isSlave():
        syncWithMaster()                       # ф╗ОцЬНхКбхЩиф╕Оф╕╗цЬНхКбхЩихРМцнецХ░цНо

# цФпч║┐я╝ЪхУихЕ╡чЫСцОзф╕ОцХЕщЪЬш╜мчз╗
function handleSentinel():
    if server.isSentinel():
        monitorMasterStatus()                  # хУихЕ╡цибх╝Пя╝МчЫСцОзф╕╗цЬНхКбхЩичК╢цАБ
        if masterIsDown():
            performFailover()                  # ф╕╗цЬНхКбхЩицХЕщЪЬцЧ╢я╝МшзжхПСцХЕщЪЬш╜мчз╗

function performFailover():
    newMaster = selectNewMaster()              # щАЙф╕╛цЦ░чЪДф╕╗цЬНхКбхЩи
    promoteSlaveToMaster(newMaster)            # цПРхНЗф╗ОцЬНхКбхЩиф╕║ф╕╗цЬНхКбхЩи
    notifyOtherSlavesAndClients(newMaster)     # щАЪчЯехЕ╢ф╗Цф╗ОцЬНхКбхЩихТМховцИ╖члп

# цФпч║┐я╝ЪщЫЖч╛дцибх╝Пф╕ЛчЪДхИЖчЙЗф╕ОцХ░цНош╖пчФ▒
function handleCluster():
    if server.isCluster():
        for each command in client.commands:
            keyHashSlot = computeKeyHashSlot(command.key) # шобчоЧ key чЪДхУИх╕Мцз╜
            if hashSlotIsLocal(keyHashSlot):
                executeCommandLocally(command)   # хжВцЮЬхУИх╕Мцз╜х▒Юф║ОцЬмшКВчВ╣я╝МцЙзшбМхС╜ф╗д
            else:
                forwardCommandToCorrectNode(command) # х░ЖхС╜ф╗дш╜мхПСхИ░цнгчбочЪДшКВчВ╣
